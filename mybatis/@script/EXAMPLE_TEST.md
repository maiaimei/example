PostgreSQL版本:

```sql
-- 建表语句
DROP TABLE "EXAMPLE_TEST";
CREATE TABLE "EXAMPLE_TEST" (
    "ID" NUMERIC(22) PRIMARY KEY,
    "NAME" VARCHAR(100),
    "INTEGER_LIST" INTEGER[],
    "BIG_DECIMAL_LIST" NUMERIC(22)[],
    "STRING_LIST" VARCHAR[],
    "INTEGER_ARRAY" INTEGER[],
    "BIG_DECIMAL_ARRAY" NUMERIC(22)[],
    "STRING_ARRAY" VARCHAR[], 
    "CREATED_AT" TIMESTAMP
);

DROP TABLE "JSONB_TEST";
CREATE TABLE "JSONB_TEST" (
    "ID" NUMERIC(22) PRIMARY KEY,
    "DATA" JSONB
);
```
